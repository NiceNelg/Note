# 面向对象编程

### 基本形式

```go
type Person struct {
    id int
    name string
    sex byte
}

func (this *Person) say(msg string) {
    fmt.PrintIn(this.name, " say：", msg)
}

func main() {
    nelg := Person{name:"Nelg"}
    nelg.say("Hello!")
}
```

### 继承

```go
type Person struct {
    id int
    name string
    sex byte
}

type Student struct {
    Person
    desc string
}

func (this *Person) say(msg string) {
    fmt.PrintIn(this.name, " say：", msg)
}

func main() {
    nelg := Student{Person{name:"Nelg"}}
    nelg.say("Hello!")
}
```

### 多态

```go
type behavior interface {
    sayHello()
}

type Person struct {
    id int
    name string
    sex byte
}

func (this *Person) say() {
    fmt.PrintIn(this.name, " say：I am person")
}

type Student struct {
    Person
    desc string
}

func (this *Student) say() {
    fmt.PrintIn(this.name, " say：I am student")
}

func main() {
    var inter behavior
    
    zq := Person{name:"zq"}
    inter = zq
    inter.say()
    //zq say：I am person
    
    nelg := Student{Person{name:"Nelg"}}
    inter = nelg
    inter.say()
    //nelg say：I am student
}
```

> 空接口可以赋值任何类型，如
>
> ```go
> var i interface{}
> 
> i = 1
> i = '2'
> i = "Nelg"
> ```
